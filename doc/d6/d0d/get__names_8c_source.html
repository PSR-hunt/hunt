00129 int get_remote_name(char *his_address) {
00130         char *his_name;
00131         char *his_machine_name;
00132         char *ptr;
00133         struct hostent *hp;
00134 
00135         /* check for, and strip out, the machine name of the target */
00136         for (ptr = his_address; *ptr != '\0' && *ptr != '@' && *ptr != ':'
00137                         && *ptr != '!' && *ptr != '.'; ptr++) {
00138                 continue;
00139         }
00140         if (*ptr == '\0') {
00141                 /* this is a local to local talk */
00142                 his_name = his_address;
00143                 his_machine_name = my_machine_name;
00144         } else {
00145                 if (*ptr == '@') {
00146                         /* user@host */
00147                         his_name = his_address;
00148                         his_machine_name = ptr + 1;
00149                 } else {
00150                         /* host.user or host!user or host:user */
00151                         his_name = ptr + 1;
00152                         his_machine_name = his_address;
00153                 }
00154                 *ptr = '\0';
00155         }
00156         /* Load these useful values into the standard message header */
00157         (void) strncpy(msg.r_name, his_name, NAME_SIZE);
00158         msg.r_name[NAME_SIZE - 1] = '\0';
00159 
00160         /* if he is on the same machine, then simply copy */
00161         if (memcmp((char *) &his_machine_name, (char *) &my_machine_name,
00162                                         sizeof(his_machine_name)) == 0) {
00163                 memcpy(&his_machine_addr, &my_machine_addr,
00164                                 sizeof(his_machine_name));
00165         }
00166         else {
00167                 /* look up the address of the recipient's machine */
00168                 hp = gethostbyname(his_machine_name);
00169                 if (hp == (struct hostent *) 0) {
00170                         return 0; /* unknown host */
00171                 }
00172                 memcpy(&his_machine_addr, hp->h_addr, hp->h_length);
00173         }
00174         return 1;
00175 }

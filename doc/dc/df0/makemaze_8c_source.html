00153 void dig_maze(int x, int y) {
00154         int tx, ty;
00155         int i, j;
00156         int order[4];
00157 #define MNORTH  0x1 //1 in binary.
00158 #define MSOUTH  0x2
00159 #define MEAST   0x4
00160 #define MWEST   0x8
00161 
00162         tx = ty = 0;
00163         Maze[y][x] = SPACE;
00164         order[0] = MNORTH;
00165         for (i = 1; i < 4; i++) {
00166                 j = rand_num(i + 1);
00167                 order[i] = order[j];
00168                 order[j] = 0x1 << i; //Shifts 0x1 on the left by i positions.
00169         }
00170         for (i = 0; i < 4; i++) {
00171                 switch (order[i]) {
00172                 case MNORTH:
00173                         tx = x;
00174                         ty = y - 2;
00175                         break;
00176                 case MSOUTH:
00177                         tx = x;
00178                         ty = y + 2;
00179                         break;
00180                 case MEAST:
00181                         tx = x + 2;
00182                         ty = y;
00183                         break;
00184                 case MWEST:
00185                         tx = x - 2;
00186                         ty = y;
00187                         break;
00188                 }
00189                 if (tx < 0 || ty < 0 || tx >= WIDTH || ty >= HEIGHT){
00190                         continue;
00191                 }
00192                 if (Maze[ty][tx] == SPACE){
00193                         continue;
00194                 }
00195                 Maze[(y + ty) / 2][(x + tx) / 2] = SPACE;
00196                 dig_maze(tx, ty);
00197         }
00198 }

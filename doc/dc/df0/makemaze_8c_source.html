00158 void dig_maze(int x, int y) {
00159         int tx, ty;
00160         int i, j;
00161         int order[4];
00162 #define MNORTH  0x1 
00163 #define MSOUTH  0x2 
00164 #define MEAST   0x4 
00165 #define MWEST   0x8 
00167         tx = ty = 0;
00168         Maze[y][x] = SPACE;
00169         order[0] = MNORTH;
00170         for (i = 1; i < 4; i++) {
00171                 j = rand_num(i + 1);
00172                 order[i] = order[j];
00173                 order[j] = 0x1 << i; /* Shifts 0x1 on the left by i positions. [PSR] */
00174         }
00175         for (i = 0; i < 4; i++) {
00176                 switch (order[i]) {
00177                 case MNORTH:
00178                         tx = x;
00179                         ty = y - 2;
00180                         break;
00181                 case MSOUTH:
00182                         tx = x;
00183                         ty = y + 2;
00184                         break;
00185                 case MEAST:
00186                         tx = x + 2;
00187                         ty = y;
00188                         break;
00189                 case MWEST:
00190                         tx = x - 2;
00191                         ty = y;
00192                         break;
00193                 }
00194                 if (tx < 0 || ty < 0 || tx >= WIDTH || ty >= HEIGHT){
00195                         continue;
00196                 }
00197                 if (Maze[ty][tx] == SPACE){
00198                         continue;
00199                 }
00200                 Maze[(y + ty) / 2][(x + tx) / 2] = SPACE;
00201                 dig_maze(tx, ty);
00202         }
00203 }

                order[i] = order[j];

        }
        for (i = 0; i < 4; i++) {
                switch (order[i]) {
                case MNORTH:
                        tx = x;
                        ty = y - 2;
                        break;
                case MSOUTH:
                        tx = x;
                        ty = y + 2;
                        break;
                case MEAST:
                        tx = x + 2;
                        ty = y;
                        break;
                case MWEST:
                        tx = x - 2;
                        ty = y;
                        break;

                        continue;

                if (Maze[ty][tx] == SPACE){
                        continue;
                }

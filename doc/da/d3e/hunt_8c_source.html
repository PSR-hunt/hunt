00490 SOCKET * list_drivers() {
00491         int option;
00492         unsigned short msg; 
00493         unsigned short port_num; 
00494         static SOCKET test;
00495         int test_socket;
00500         unsigned int namelen;
00501         char local_name[MAXHOSTNAMELEN + 1];
00502         static int initial = true;
00503         static struct in_addr local_address;
00504         struct hostent *hp;
00505 # ifdef BROADCAST
00506         static int brdc;
00507         static SOCKET *brdv;
00508 # else
00509         unsigned long local_net; 
00510 # endif
00511         int i;
00512         static SOCKET *listv;
00513         static unsigned int listmax;
00514         unsigned int listc;
00515         struct pollfd set[1];
00516 
00517         if (initial) { /* do one time initialization */
00518 # ifndef BROADCAST
00519                 sethostent(1); /* don't bother to close host file */
00520 # endif
00521                 if (gethostname(local_name, sizeof local_name) < 0) {
00522                         leavex(1, "Sorry, I have no name.");
00523                         /* NOTREACHED */
00524                 }
00525                 local_name[sizeof(local_name) - 1] = '\0';
00526                 if ((hp = gethostbyname(local_name)) == NULL) {
00527                         leavex(1, "Can't find myself.");
00528                         /* NOTREACHED */
00529                 }
00530                 local_address = * ((struct in_addr *) hp->h_addr);
00531 
00532                 listmax = 20;
00533                 listv = (SOCKET *) malloc(listmax * sizeof (SOCKET));
00534                 if (listv == NULL) {
00535                         leave(1, "Out of memory!");
00536                 }
00537         } else if (Sock_host != NULL) {
00538                 return listv; /* address already valid */
00539         }
00540 
00541         test_socket = socket(SOCK_FAMILY, SOCK_DGRAM, 0); //Creation of an UDP socket
00542         if (test_socket < 0) {
00543                 leave(1, "socket system call failed");
00544                 /* NOTREACHED */
00545         }
00546         test.sin_family = SOCK_FAMILY;
00547         test.sin_port = htons(Test_port);
00548         listc = 0;
00549 
00550         if (Sock_host != NULL) { /* explicit host given */
00551                 if ((hp = gethostbyname(Sock_host)) == NULL) {
00552                         leavex(1, "Unknown host");
00553                         /* NOTREACHED */
00554                 }
00555                 test.sin_addr = *((struct in_addr *) hp->h_addr);
00556                 msg = htons(C_TESTMSG());
00557                 (void) sendto(test_socket, (char *) &msg, sizeof msg, 0,
00558                                 (struct sockaddr *) &test, DAEMON_SIZE);
00559         } else {
00560 
00561                 if (!initial) {
00562                         /* favor host of previous session by broadcasting to it first */
00563                         test.sin_addr = Daemon.sin_addr;
00564                         msg = htons(C_PLAYER); /* Must be playing! */
00565                         (void) sendto(test_socket, (char *) &msg, sizeof msg, 0,
00566                                         (struct sockaddr *) &test, DAEMON_SIZE);
00567                 }
00568 
00569 # ifdef BROADCAST
00570                 if (initial) {
00571                         brdc = broadcast_vec((void *) &brdv);
00572                 }
00573 
00574 # ifdef SO_BROADCAST
00575                 /* Sun's will broadcast even though this option can't be set */
00576                 option = 1;
00577                 if (setsockopt(test_socket, SOL_SOCKET, SO_BROADCAST,
00578                                                 &option, sizeof option) < 0) {
00579                         leave(1, "setsockopt broadcast");
00580                         /* NOTREACHED */
00581                 }
00582 # endif
00583 
00584                 /* send broadcast packets on all interfaces */
00585                 msg = htons(C_TESTMSG());
00586                 for (i = 0; i < brdc; i++) {
00587                         test.sin_addr = brdv[i].sin_addr;
00588                         if (sendto(test_socket, (char *) &msg, sizeof msg, 0,
00589                                                         (struct sockaddr *) &test, DAEMON_SIZE) < 0) {
00590                                 leave(1, "sendto");
00591                                 /* NOTREACHED */
00592                         }
00593                 }
00594                 test.sin_addr = local_address;
00595                 if (sendto(test_socket, (char *) &msg, sizeof msg, 0,
00596                                                 (struct sockaddr *) &test, DAEMON_SIZE) < 0) {
00597                         leave(1, "sendto");
00598                         /* NOTREACHED */
00599                 }
00600 # else /* !BROADCAST */
00601                 /* loop thru all hosts on local net and send msg to them. */
00602                 msg = htons(C_TESTMSG());
00603                 local_net = inet_netof(local_address);
00604                 sethostent(0); /* rewind host file */
00605                 while (hp = gethostent()) {
00606                         if (local_net == inet_netof(* ((struct in_addr *) hp->h_addr))) {
00607                                 test.sin_addr = * ((struct in_addr *) hp->h_addr);
00608                                 (void) sendto(test_socket, (char *) &msg, sizeof msg, 0,
00609                                                 (struct sockaddr *) &test, DAEMON_SIZE);
00610                         }
00611                 }
00612 # endif
00613         }
00614 
00615         namelen = DAEMON_SIZE;
00616         errno = 0;
00617         set[0].fd = test_socket;
00618         set[0].events = POLLIN;
00619         for (;;) {
00620                 if (listc + 1 >= listmax) {
00621                         listmax += 20;
00622                         listv = (SOCKET *) realloc((char *) listv,
00623                                         listmax * sizeof(SOCKET));
00624                         if (listv == NULL) {
00625                                 leave(1, "Out of memory!");
00626                         }
00627                 }
00628 
00629                 if (poll(set, 1, 1000) == 1 &&
00630                                 recvfrom(test_socket, (char *) &port_num, sizeof(port_num),
00631                                                 0, (struct sockaddr *) &listv[listc], &namelen) > 0) {
00632                         /*
00633                          * Note that we do *not* convert from network to host
00634                          * order since the port number *should* be in network
00635                          * order:
00636                          */
00637                         for (i = 0; (unsigned)i < listc; i += 1) {
00638                                 if (listv[listc].sin_addr.s_addr
00639                                                 == listv[i].sin_addr.s_addr) {
00640                                         break;
00641                                 }
00642                         }
00643                         if ((unsigned)i == listc) {
00644                                 listv[listc++].sin_port = port_num;
00645                         }
00646                         continue;
00647                 }
00648 
00649                 if (errno != 0 && errno != EINTR) {
00650                         leave(1, "poll/recvfrom");
00651                         /* NOTREACHED */
00652                 }
00653 
00654                 /* terminate list with local address */
00655                 listv[listc].sin_family = SOCK_FAMILY;
00656                 listv[listc].sin_addr = local_address;
00657                 listv[listc].sin_port = htons(0);
00658 
00659                 (void) close(test_socket);
00660                 initial = false;
00661                 return listv;
00662         }
00663 
00664 }

00247 static unsigned char getchr() {
00248         struct pollfd set[2];
00249         int nfds;
00250 
00251         set[0].fd = Socket;
00252         set[0].events = POLLIN;
00253         set[1].fd = STDIN;
00254         set[1].events = POLLIN;
00255 
00256         while (1) {
00257                 do {
00258                         errno = 0;
00259                         nfds = poll(set, 2, INFTIM);
00260                 } while (nfds <= 0 && errno == EINTR);
00261 
00262                 if (set[1].revents && POLLIN)
00263                         send_stuff();
00264                 if (!(set[0].revents & POLLIN))
00265                         continue;
00266                 icnt = read(Socket, ibuf, sizeof ibuf);
00267                 if (icnt < 0) {
00268                         bad_con();
00269                         /* NOTREACHED */
00270                 }
00271                 if (icnt == 0)
00272                         continue;
00273                 iptr = ibuf;
00274                 icnt--;
00275                 return *iptr++;
00276         }
00277         //This point should never be reached
00278         return -1;

00284 static void send_stuff() {
00285         int count;
00286         char *sp, *nsp;
00287         static char inp[sizeof Buf];
00288 
00289         count = read(STDIN, Buf, sizeof Buf);
00290         if (count <= 0)
00291                 return;
00292         if (nchar_send <= 0 && !no_beep) {
00293                 dbg_write(1, "\7", 1);
00294                 /* CTRL('G') */
00295                 return;
00296         }
00297 
00298         /*
00299          * look for 'q'uit commands; if we find one,
00300          * confirm it.  If it is not confirmed, strip
00301          * it out of the input
00302          */
00303         Buf[count] = '\0';
00304         nsp = inp;
00305         for (sp = Buf; *sp != '\0'; sp++)
00306                 if ((*nsp = map_key[(int) *sp]) == 'q')
00307                         intr(0);
00308                 else
00309                         nsp++;
00310         count = nsp - inp;
00311         if (count) {
00312 # ifdef OTTO
00313                 Otto_count += count;
00314 # endif
00315                 nchar_send -= count;
00316                 if (nchar_send < 0)
00317                         count += nchar_send;
00318                 dbg_write(Socket, inp, count);
00319         }
